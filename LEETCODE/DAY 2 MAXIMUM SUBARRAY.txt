class Solution {
    public int maxSubArray(int[] nums) {
        int n=nums.length,max=Integer.MIN_VALUE,curr=0;
        if(n==1)return nums[0];
        for(int i=0;i<n;i++){
           curr+=nums[i];
          if(curr>max){
            max=curr;
          }
          if(curr<0)curr=0;
         
        }
        return max;
    }
     
}
